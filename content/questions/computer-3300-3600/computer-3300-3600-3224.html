pada OpenSuSE, seseorang mengedit file yang sesuai di ~ / .config / openbox, yaitu ~/.config/openbox/lxde-rc.xml, dan dapat memperbarui konfigurasi yang sedang berjalan dengan bagian openbox-lxde --reconfigure

\n\n

I added the following in the <keyboard>, 

<!-- Maximize commands -->\n<keybind key=\"W-Up\">\n    <action name=\"ToggleMaximizeFull\" />\n</keybind>\n\n<keybind key=\"W-S-Up\">\n    <action name=\"ToggleMaximizeVert\" />\n</keybind>\n\n<keybind key=\"W-S-Down\">\n    <action name=\"ToggleMaximizeHorz\" />\n</keybind>\n\n<!-- application execute commands -->\n<keybind key=\"W-f\">\n    <action name=\"Execute\">\n        <command>chromium</command>\n    </action>\n</keybind>\n\n<keybind key=\"W-t\">\n    <action name=\"Execute\">\n        <command>lxterminal</command>\n    </action>\n</keybind>\n\n<!-- logout -->\n<keybind key=\"C-A-S-Delete\">\n    <action name=\"Exit\">\n        <prompt>no</prompt>\n    </action>\n</keybind>\n\n<!-- window movement -->\n<keybind key=\"W-A-Up\">\n    <action name=\"MoveToEdgeNorth\"/>\n</keybind>\n\n<keybind key=\"W-A-Down\">\n    <action name=\"MoveToEdgeSouth\"/>\n</keybind>\n\n<keybind key=\"W-A-Left\">\n    <action name=\"MoveToEdgeWest\"/>\n</keybind>\n\n<keybind key=\"W-A-Right\">\n    <action name=\"MoveToEdgeEast\"/>\n</keybind>\n
