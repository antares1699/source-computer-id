Untuk kelengkapan, inilah perubahan untuk tombol 1..9: 

awful.key({ modkey }, \"#\" .. i + 9,\n    function ()\n        for screen = 1, screen.count() do\n            local tag = awful.tag.gettags(screen)[i]\n            if tag then\n                awful.tag.viewonly(tag)\n            end\n        end\n    end\n),\n

Dan untuk menangani klik pada daftar tag: 

mytaglist.buttons = awful.util.table.join(\n  awful.button({ }, 1, function(tag)\n    local i = awful.tag.getidx(tag)\n    for screen = 1, screen.count() do\n        local tag = awful.tag.gettags(screen)[i]\n        if tag then\n           awful.tag.viewonly(tag)\n        end\n    end\n  end),\n[...]\n
