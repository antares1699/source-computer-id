Anda sebenarnya dapat menggunakan aturan firewall berikut pada firmware dd-wrt untuk memaksa pengguna menggunakan server DNS yang ditentukan (dalam pengaturan router) 

iptables -t nat -A PREROUTING -p udp -i br0 --dport 53 -j DNAT --to $(nvram get lan_ipaddr)\niptables -t nat -A PREROUTING -p tcp -i br0 --dport 53 -j DNAT --to $(nvram get lan_ipaddr)\n
