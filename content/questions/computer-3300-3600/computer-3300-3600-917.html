Berikut adalah rutinitas VBA sederhana yang membuat bagan Sebar XY dari data yang disusun dengan Y di kolom pertama dan satu atau beberapa set X di data berikutnya. Pilih rentang data, atau pilih satu sel dalam rentang itu, dan jalankan kodenya. 

Sub ChartFromInput_YXX()\n    ' assumes first column of input range has Y values\n    ' assumes subsequent columns have X values\n    ' assumes first row of input range has series\n\n    Dim rngDataSource As Range\n    Dim iDataRowsCt As Long\n    Dim iDataColsCt As Integer\n    Dim iSrsIx As Integer\n    Dim chtChart As Chart\n    Dim srsNew As Series\n\n    If Not TypeName(Selection) = \"Range\" Then\n        '' Doesn't work if no range is selected\n        MsgBox \"Please select a data range and try again.\", _\n            vbExclamation, \"No Range Selected\"\n    Else\n        Set rngDataSource = Selection\n        If rngDataSource.Cells.Count = 1 Then\n            ' expand to data range including active cell\n            Set rngDataSource = rngDataSource.CurrentRegion\n        End If\n        With rngDataSource\n            iDataRowsCt = .Rows.Count\n            iDataColsCt = .Columns.Count\n        End With\n\n        '' Create the chart\n        Set chtChart = ActiveSheet.Shapes.AddChart.Chart\n\n        With chtChart\n            .ChartType = xlXYScatterLines\n\n            '' Remove any series created with the chart\n            Do Until .SeriesCollection.Count = 0\n                .SeriesCollection(1).Delete\n            Loop\n\n            For iSrsIx = 1 To iDataColsCt - 1\n                '' Add each series\n                Set srsNew = .SeriesCollection.NewSeries\n                With srsNew\n                    .Name = rngDataSource.Cells(1, 1 + iSrsIx)\n                    .Values = rngDataSource.Cells(2, 1) _\n                        .Resize(iDataRowsCt - 1, 1)\n                    .XValues = rngDataSource.Cells(2, 1 + iSrsIx) _\n                        .Resize(iDataRowsCt - 1, 1)\n                End With\n            Next\n        End With\n    End If\nEnd Sub\n
