=TRIM(MID(SUBSTITUTE($A1,\" \",REPT(\" \",999)),COLUMNS($A:A)*999-998,999)) & \" \" & TRIM(MID(SUBSTITUTE($A1,\" \",REPT(\" \",999)),COLUMNS($A:B)*999-998,999))\n
