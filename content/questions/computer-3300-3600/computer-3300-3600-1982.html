Masalah yang ditemukan: psmouse / elantech.c hanya berasumsi, bahwa protokol perangkat keras V4 tidak memiliki tombol kanan sama sekali.

\n\n

Perbaiki itu sepele: 

--- elantech.c.orig 2013-06-14 00:52:08.000000000 +0700\n+++ elantech.c  2013-07-15 12:45:52.000000000 +0700\n@@ -486,6 +486,7 @@ elantech_input_sync_v4\n    unsigned char *packet = psmouse->packet;\n\n    input_report_key(dev, BTN_LEFT, packet[0] & 0x01);\n+   input_report_key(dev, BTN_RIGHT, packet[0] & 0x02);\n    input_mt_report_pointer_emulation(dev, true);\n    input_sync(dev);\n }\n@@ -1019,7 +1018,7 @@ elantech_set_input_params { case 4: }\n        }\n        /* v4 is clickpad, with only one button. */\n        __set_bit(INPUT_PROP_BUTTONPAD, dev->propbit);\n-       __clear_bit(BTN_RIGHT, dev->keybit);\n+       //__clear_bit(BTN_RIGHT, dev->keybit);\n        __set_bit(BTN_TOOL_QUADTAP, dev->keybit);\n        /* For X to recognize me as touchpad. */\n        input_set_abs_params(dev, ABS_X, x_min, x_max, 0, 0);\n

Solusi siap pakai: 

wget http://datacompboy.ru/psmouse-elantech-v7right.tar.bz2\ntar xfj psmouse-elantech-v7right.tar.bz2\nsudo mv psmouse-elantech-v7right /usr/src/\nsudo dkms install -m psmouse -v elantech-v7right\nsudo modprobe -r psmouse\nsudo modprobe psmouse\n
