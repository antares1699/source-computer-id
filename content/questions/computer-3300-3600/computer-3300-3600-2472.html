Di sini saya mereproduksi skrip PowerShell Nils Schimmelmann untuk membuat dua tingkatan\nRuang Penyimpanan, untuk SSD dan HDD: 

#Variables\n$StoragePoolName = \"My Storage Pool\"\n$TieredSpaceName = \"My Tiered Space\"\n$ResiliencySetting = \"Simple\"\n$SSDTierName = \"SSDTier\"\n$HDDTierName = \"HDDTier\"\n\n#List all disks that can be pooled and output in table format (format-table)\nGet-PhysicalDisk -CanPool $True | ft FriendlyName,OperationalStatus,Size,MediaType\n\n#Store all physical disks that can be pooled into a variable, $PhysicalDisks\n$PhysicalDisks = (Get-PhysicalDisk -CanPool $True | Where MediaType -NE UnSpecified)       \n\n#Create a new Storage Pool using the disks in variable $PhysicalDisks with a name of My Storage Pool\n$SubSysName = (Get-StorageSubSystem).FriendlyName\nNew-StoragePool -PhysicalDisks $PhysicalDisks -StorageSubSystemFriendlyName $SubSysName -FriendlyName $StoragePoolName\n\n#View the disks in the Storage Pool just created\nGet-StoragePool -FriendlyName $StoragePoolName | Get-PhysicalDisk | Select FriendlyName, MediaType\n\n#Create two tiers in the Storage Pool created. One for SSD disks and one for HDD disks\n$SSDTier = New-StorageTier -StoragePoolFriendlyName $StoragePoolName -FriendlyName $SSDTierName -MediaType SSD\n$HDDTier = New-StorageTier -StoragePoolFriendlyName $StoragePoolName -FriendlyName $HDDTierName -MediaType HDD\n\n#Identify tier sizes within this storage pool\n$SSDTierSizes = (Get-StorageTierSupportedSize -FriendlyName $SSDTierName -ResiliencySettingName $ResiliencySetting).TierSizeMax\n$HDDTierSizes = (Get-StorageTierSupportedSize -FriendlyName $HDDTierName -ResiliencySettingName $ResiliencySetting).TierSizeMax \n\n#Create a new virtual disk in the pool with a name of TieredSpace using the SSD and HDD tiers\nNew-VirtualDisk -StoragePoolFriendlyName $StoragePoolName -FriendlyName $TieredSpaceName -StorageTiers $SSDTier, $HDDTier -StorageTierSizes $SSDTierSizes, $HDDTierSizes -ResiliencySettingName $ResiliencySetting  -AutoWriteCacheSize -AutoNumberOfColumns\n\n#Alternatively try adjusting the sizes manually:\n#New-VirtualDisk -StoragePoolFriendlyName $StoragePoolName -FriendlyName $TieredSpaceName -StorageTiers @($SSDTier,$HDDTier) -StorageTierSizes @(228GB,1.816TB) -ResiliencySettingName $ResiliencySetting -AutoWriteCacheSize -AutoNumberOfColumns\n
