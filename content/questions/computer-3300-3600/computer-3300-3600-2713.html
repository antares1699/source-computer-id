sudo git fetch origin;\nsudo git checkout master;\nsudo git pull;\n

--- jika port 80 terbuka 

echo Starting Meteor;\nexport LC_ALL=C;\nexport ROOT_URL=$ROOT_URL;\nsudo meteor --port 80;\n

--- lain 

echo Meteor already running;\n

Kemudian sebagai ceri di atas karena Meteor adalah proses yang berjalan lama, bagaimana cara membuatnya berjalan di latar belakang dan keluar dari skrip? (Saya sudah mencoba nohup, &, tetapi saya tidak tahu apa praktik terbaiknya?)
