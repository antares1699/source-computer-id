Anda mengatakan "dengan tangan" yang membingungkan sebagian orang, saya kira. Anda tidak benar-benar mengatakan bahwa Anda ingin menulis PDF mentah, bukan? Di bawah ini adalah kode yang menggunakan open source perpustakaan iTextSharp (5.1.1.0). Atur variabel FolderWithImages ke folder Anda yang berisi gambar dan PdfFileName ke PDF yang ingin Anda keluarkan dan itu akan mengambil semua JPG di folder dan membuat PDF. Kode ini sangat sederhana tetapi Anda dapat melakukan banyak hal seperti mengubah ukuran, penskalaan, dll. Ada banyak kode di luar sana untuk iTextSharp dan proyek induknya, iText. 

using System;\nusing System.ComponentModel;\nusing System.IO;\nusing System.Windows.Forms;\nusing iTextSharp.text;\nusing iTextSharp.text.pdf;\n\nnamespace WindowsFormsApplication2\n{\n    public partial class Form1 : Form\n    {\n        public Form1()\n        {\n            InitializeComponent();\n        }\n\n        private void Form1_Load(object sender, EventArgs e)\n        {\n            //The folder containing our images\n            string FolderWithImages = Environment.GetFolderPath(Environment.SpecialFolder.Desktop);\n            //The PDF that we will output\n            string PdfFileName = Path.Combine(Environment.GetFolderPath(Environment.SpecialFolder.Desktop), \"ImagesCombined.pdf\");\n\n            //Create a basic stream to write to\n            using (FileStream fs = new FileStream(PdfFileName, FileMode.Create, FileAccess.Write, FileShare.None))\n            {\n                //Create a new PDF document\n                using (Document doc = new Document())\n                {\n                    //Bind a the document to the stream\n                    using (PdfWriter w = PdfWriter.GetInstance(doc, fs))\n                    {\n                        //Open our document for writing\n                        doc.Open();\n\n                        //Will hold an instance of our image\n                        iTextSharp.text.Image img;\n\n                        //Grab all JPGs from the given folder and loop through them\n                        string[] Images = Directory.GetFiles(FolderWithImages, \"*.jpg\", SearchOption.TopDirectoryOnly);\n                        foreach (string i in Images)\n                        {\n                            //Get the JPG as an iTextSharp \"image\"\n                            img = iTextSharp.text.Image.GetInstance(i);\n\n                            //Tell the image that when placed we want it at (0,0)\n                            img.SetAbsolutePosition(0, 0);\n\n                            //Tell the system that the next \"page\" that we add should be the dimension of the image\n                            doc.SetPageSize(new iTextSharp.text.Rectangle(0, 0, img.Width, img.Height));\n\n                            //Add a new blank page\n                            doc.NewPage();\n\n                            //Put the image on the blank page\n                            doc.Add(img);\n                        }\n\n                        //Close our output PDF\n                        doc.Close();\n                    }\n                }\n            }\n            this.Close();\n        }\n    }\n}\n
