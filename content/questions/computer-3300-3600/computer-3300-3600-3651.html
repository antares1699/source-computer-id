rm -rf ~/VirtualBox\\ VMs ~/Library/VirtualBox\nbrew cask uninstall virtualbox\nbrew cask install virtualbox\nfor bin in VirtualBox VirtualBoxVM VBoxNetAdpCtl VBoxNetDHCP VBoxNetNAT VBoxHeadless; do\n    sudo chmod u+s \"/Applications/VirtualBox.app/Contents/MacOS/${bin}\"\ndone\n
