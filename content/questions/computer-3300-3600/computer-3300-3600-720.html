Untuk menjelaskan pada jawabannya dari Tiago Cardoso , Saya menambahkan kondisi untuk memeriksa untuk memastikan tinggi baris maksimum tidak terlampaui saat meningkatkan tinggi baris dengan variabel faktor pengubah ukuran. 

Sub FixRowHeight()\n    Dim oRange As Excel.Range\n    Dim oRow As Excel.Range\n    Dim dblFactor As Double\n\n    Set oRange = Sheets(1).Range(\"A1:O367\")\n    dblFactor = 12\n\n    For Each oRow In oRange.Rows\n        If oRow.RowHeight + dblFactor <= 409.5 Then\n            oRow.RowHeight = oRow.RowHeight + dblFactor\n        Else\n            oRow.RowHeight = 409.5\n        End If\n    Next oRow\nEnd Sub\n
