Dengan asumsi data Anda ada di sel B4

Sel C4

=LEFT(B4,SEARCH(" ",B4,SEARCH(" ",B4)+1)-1)

Sel D4

=LEFT((RIGHT(B4,LEN(B4)-LEN(C4)-1)),SEARCH(" ",(RIGHT(B4,LEN(B4)-LEN(C4)-1)),SEARCH(" ",(RIGHT(B4,LEN(B4)-LEN(C4)-1)))+1)-1)

Sel E4

=RIGHT(B4,LEN(B4)-(LEN(C4)+LEN(D4)+2))
