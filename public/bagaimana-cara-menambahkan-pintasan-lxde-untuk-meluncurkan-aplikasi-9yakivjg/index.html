Dokumentasi lebih lanjut ada di sini:

  • http://openbox.org/wiki/Help:Bindings (dari tautan Christoffer)
  • http://openbox.org/wiki/Help:Actions - daftar aksi yang sangat bagus

pada OpenSuSE, seseorang mengedit file yang sesuai di ~ / .config / openbox, yaitu ~/.config/openbox/lxde-rc.xml, dan dapat memperbarui konfigurasi yang sedang berjalan dengan bagian openbox-lxde --reconfigure

I added the following in the <keyboard>,

<!-- Maximize commands -->
<keybind key="W-Up">
    <action name="ToggleMaximizeFull" />
</keybind>

<keybind key="W-S-Up">
    <action name="ToggleMaximizeVert" />
</keybind>

<keybind key="W-S-Down">
    <action name="ToggleMaximizeHorz" />
</keybind>

<!-- application execute commands -->
<keybind key="W-f">
    <action name="Execute">
        <command>chromium</command>
    </action>
</keybind>

<keybind key="W-t">
    <action name="Execute">
        <command>lxterminal</command>
    </action>
</keybind>

<!-- logout -->
<keybind key="C-A-S-Delete">
    <action name="Exit">
        <prompt>no</prompt>
    </action>
</keybind>

<!-- window movement -->
<keybind key="W-A-Up">
    <action name="MoveToEdgeNorth"/>
</keybind>

<keybind key="W-A-Down">
    <action name="MoveToEdgeSouth"/>
</keybind>

<keybind key="W-A-Left">
    <action name="MoveToEdgeWest"/>
</keybind>

<keybind key="W-A-Right">
    <action name="MoveToEdgeEast"/>
</keybind>
